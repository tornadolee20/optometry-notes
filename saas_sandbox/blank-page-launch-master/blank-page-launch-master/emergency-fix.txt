import { serve } from "https://deno.land/std@0.168.0/http/server.ts";
import "https://deno.land/x/xhr@0.1.0/mod.ts";

const corsHeaders = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type',
};

serve(async (req) => {
  if (req.method === 'OPTIONS') {
    return new Response(null, { headers: corsHeaders });
  }

  try {
    console.log('=== 開始處理評論生成請求 ===');
    
    const requestBody = await req.json();
    console.log('✅ JSON解析成功');
    
    // 簡化參數提取
    const storeName = requestBody.storeName || '測試店家';
    const address = requestBody.address || '測試地址';
    const area = requestBody.area || '該地區';
    const keywords = requestBody.keywords || [];
    const customFeelings = requestBody.customFeelings || [];
    
    console.log('參數:', { storeName, address, area, keywords, customFeelings });
    
    // 基本驗證
    if (keywords.length === 0 && customFeelings.length === 0) {
      return new Response(
        JSON.stringify({ error: '至少需要一個關鍵字或自訂感受' }),
        { status: 400, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
      );
    }
    
    // 檢查API金鑰
    const openAIApiKey = Deno.env.get('OPENAI_API_KEY');
    if (!openAIApiKey) {
      return new Response(
        JSON.stringify({ error: '找不到 OpenAI API 金鑰' }),
        { status: 500, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
      );
    }
    
    // 合併所有關鍵字
    const allKeywords = [...keywords, ...customFeelings];
    
    // 簡化的系統提示
    const systemPrompt = `請撰寫一則 Google 評論：
1. 字數範圍：150-200字
2. 地區：${area}
3. 關鍵字：${allKeywords.join('、')}
4. 風格：自然、真實
5. 語氣：正面積極`;
    
    const userMessage = `請為 ${storeName} (位於${address}) 生成一則評論，包含以下關鍵字：${allKeywords.join('、')}`;
    
    console.log('🚀 發送OpenAI API請求...');
    
    // 調用OpenAI API
    const response = await fetch('https://api.openai.com/v1/chat/completions', {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${openAIApiKey}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        model: 'gpt-4o-mini',
        messages: [
          { role: 'system', content: systemPrompt },
          { role: 'user', content: userMessage }
        ],
        temperature: 0.8,
        max_tokens: 400,
      }),
    });
    
    if (!response.ok) {
      const errorData = await response.json();
      console.error('OpenAI API error:', errorData);
      return new Response(
        JSON.stringify({ error: 'OpenAI API調用失敗', details: errorData }),
        { status: 502, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
      );
    }
    
    const data = await response.json();
    const generatedText = data.choices[0].message.content.trim();
    
    console.log('✅ 評論生成成功:', generatedText.substring(0, 50) + '...');
    
    return new Response(
      JSON.stringify({ review: generatedText }),
      { headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
    );
    
  } catch (error) {
    console.error('Error:', error);
    return new Response(
      JSON.stringify({ error: error.message || '生成評論時發生錯誤' }),
      { status: 500, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }
    );
  }
});
