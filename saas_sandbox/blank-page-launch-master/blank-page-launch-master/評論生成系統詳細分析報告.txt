一、系統架構概覽
====================================================================================================

本評論生成系統採用 React/TypeScript 前端 + Supabase 後端架構，主要功能為生成符合真實使用者體驗的 Google 評論。
系統分為三大核心模組：
1. 前端評論生成邏輯（React Hooks + UI 組件）
2. 後端 API 評論生成服務（Supabase Edge Functions + OpenAI GPT-4o-mini）
3. 智慧關鍵字處理系統（關鍵字選擇 + 行業智慧識別）

```javascript
export const isOpticalStore = (storeName: string, industry?: string | null): boolean => {
  // 1. 檢查行業類別
  const opticalIndustries = ['眼鏡', '配鏡', '視光', '隱形眼鏡', '鏡框', '鏡片'];
  
  // 2. 檢查店名關鍵詞
  const opticalKeywords = [
    '眼鏡', '配鏡', '光學', '視光', '隱形', '鏡框', '鏡片', '視力', 
    'optical', 'eyewear', 'glasses', 'lens', 'optometry'
  ];
  
  return opticalIndustries.some(keyword => industry?.includes(keyword)) ||
         opticalKeywords.some(keyword => storeName.toLowerCase().includes(keyword.toLowerCase()));
};
```
